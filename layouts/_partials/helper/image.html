{{- /* Params: Resources: Where to search for images Image: Image URL Context: page context */ -}} {{- $image := .Image -}} {{- $resources := .Resources -}} {{- $context := .Context -}} {{- /* 技巧：通过获取一个不存在的资源来初始化一个“真正的空资源对象”，避开 nil 关键字 */ -}} {{- $actualResource := .Resources.Get "hugonil-placeholder" -}} {{- $permalink := ($image | safeURL) -}} {{- $width := 0 -}} {{- $height := 0 -}} {{- if $image -}} {{- $url := urls.Parse $image -}} {{- if not (in (slice "https" "http") $url.Scheme) -}} {{- /* 1. 本地图片逻辑 */ -}} {{- $localRes := false -}} {{- if $resources -}} {{- $localRes = $resources.Get $image -}} {{- end -}} {{- if not $localRes -}} {{- $localRes = resources.Get $image -}} {{- end -}} {{- if $localRes -}} {{- $actualResource = $localRes -}} {{- $permalink = $actualResource.RelPermalink -}} {{- end -}} {{- else -}} {{- /* 2. 远程图片逻辑（带 Header 和 try 容错） */ -}} {{- $options := dict "headers" (dict "User-Agent" "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/122.0.0.0 Safari/537.36" "Referer" "https://rumosky.com" ) "timeout" "10s" -}} {{- /* 使用 try 捕获 403/Media Type 错误 */ -}} {{- $tryGet := try (resources.GetRemote $image $options) -}} {{- if $tryGet.Err -}} {{- warnf "Remote image fetch failed for %q: %s" $image $tryGet.Err -}} {{- else if $tryGet.Value -}} {{- $actualResource = $tryGet.Value -}} {{- $permalink = $actualResource.Permalink -}} {{- end -}} {{- end -}} {{- /* 3. 统一处理成功获取的资源属性 */ -}} {{- if $actualResource -}} {{- /* 自动旋转 */ -}} {{- if and $context.Site.Params.imageProcessing.autoOrient (reflect.IsImageResourceProcessable $actualResource) -}} {{- $actualResource = $actualResource | images.Filter images.AutoOrient -}} {{- end -}} {{- /* 提取长宽（如果存在） */ -}} {{- if reflect.IsImageResourceWithMeta $actualResource -}} {{- $width = $actualResource.Width -}} {{- $height = $actualResource.Height -}} {{- end -}} {{- end -}} {{- end -}} {{- /* 4. 返回字典，Resource 此时要么是有效对象，要么是 Hugo 内部的空值 */ -}} {{- return (dict "Resource" $actualResource "Permalink" $permalink "Height" $height "Width" $width) -}}
